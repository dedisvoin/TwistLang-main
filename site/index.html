Возможности языка Lumen

Безопасные указатели

*auto, static, move, copy — контроль памяти без GC, как в Rust, но с простым синтаксисом.

Функции высшего порядка

Замыкания (global), функции как объекты, фабрики — мощь функционального стиля.

Макросы и метапрограммирование

#macro, #define для кодогенерации и гибкости без потери производительности.

Пространства имён

Модульность через namespace, ООП-подобный синтаксис без классов.

Глобальный контекст

global для замыканий и фабрик — сохранение состояния с первого класса.

Гибридная типизация

Статическая типизация с auto-выводом — безопасность типов без избыточности.

Примеры кода Lumen

func swap(final static arg1: *auto, final static arg2: *auto) -> Null {
    let tmp = *arg1;
    *arg1 = *arg2;
    *arg2 = tmp;
}

let a = new (10);
let b = new (20);
swap(a, b);
outln *a, *b; // 20 10
func make_multiplier(global x: Int) -> auto {
    func multiplier(y: Int) -> Int {
        ret x * y;
    }
    ret multiplier;
}

let double = make_multiplier(2);
outln double(5); // 10
func createPoint(global x: Int, global y: Int) -> auto {
    let template = namespace {
        func print() -> Null {
            outln "Point(" , x , ", " , y , ")";
        }
        func sum(other: Namespace) -> Namespace {
            ret createPoint(x + other::x, y + other::y);
        }
    };
    ret template;
}

let p1 = createPoint(10, 20);
let p2 = createPoint(30, 40);
let p3 = p1::sum(p2);
p3::print(); // Point(40, 60)
func factorial(n: Int) -> Int {
    if (n <= 1) ret 1;
    ret n * factorial(n - 1);
}

outln factorial(5); // 120
