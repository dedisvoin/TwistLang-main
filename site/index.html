Замыкания и фабрики

Функции
// Фабрика счётчиков
let counter = lambda() -> Lambda {
    namespace {
        global let count = new 0;
        
        let increment = lambda() -> Int {
            namespace {
                *count = (*count) + 1;
                let result = *count;
            }::result
        };
    }::increment
};

let next = counter();
let next2 = counter();
outln next();  // 1
outln next();  // 2
outln next2(); // 1
outln next();  // 3

Ленивые вычисления

Паттерны
struct Lazy {
    static let _computation: Lambda?;
    let _result;
    static let _evaluated: Bool = false;
    
    func force() -> auto {
        if (!this._evaluated) {
            this._result = this._computation();
            this._evaluated = true;
        }
        ret this._result;
    }
}

let heavy = Lazy(lambda() -> Int {
    namespace {
        outln "Computing...";
    }::42
});

outln heavy.force(); // Computing... 42
outln heavy.force(); // 42 (кэшировано)

Структуры и ООП

Данные
struct Person {
    static let name: String?;
    static let age: Int?;
    
    func __init__(name: String, age: Int) 
            -> typeof(Person) {
        this.name = name;
        this.age = age;
        ret this;
    }
    
    func to_string() -> String {
        ret "Name: " + this.name;
    }
}

let p1 = Person("Ivan", 19);
echo p1.to_string();

Функциональный стиль

FP
// map для массива
let map = lambda(arr: [auto, ~], 
            f: Lambda) -> [auto, ~] {
    namespace {
        let result = {};
        for (let i = 0; i < arr::len(); i = i + 1;) {
            result::push(f(arr[i]));
        };
    }::result
};

let numbers = {1, 2, 3, 4, 5};
let doubled = map(numbers, 
    lambda(x: Int) -> Int { x * 2 });
// {2, 4, 6, 8, 10}

Композиция функций

Архитектура
// Композиция функций
let compose = lambda(
    f: Func(Int)->Int, 
    g: Func(Int)->Int
) -> Func(Int)->Int {
    ret lambda(x: Int) -> Int {
        namespace {
            let intermediate = f(x);
        }::g(intermediate)
    };
};

let double = lambda(x: Int) -> Int { x * 2 };
let square = lambda(x: Int) -> Int { x * x };

let pipeline = compose(double, square);
outln pipeline(5); // (5*2)^2 = 100

Макросы
// Макрос для дебага
#macro debug_expr(expr) {
    outln "[DEBUG] " + 
           String(#expr) + 
           " = " + String(expr);
}

let x = 42;
let y = 100;
debug_expr(x + y); 
// [DEBUG] x + y = 142

Lumen IDE

Integrated Development Environment for Lumen

Lumen IDE — легкая, но мощная среда разработки, написанная на Python + PyQt6. Не требует настройки. Кастомный лексер, языковой сервер и удобное автодополнение. Интерфейс интуитивен: дерево файлов, контекстные подсказки и навигация экономят время. Встроенный отладчик помогает находить ошибки без сложной конфигурации. Мгновенная компиляция «на лету» показывает результат изменений сразу — идеально для обучения. Поддержка большого количества тем позволяет кастомизировать редактор индивидуально под каждого.
